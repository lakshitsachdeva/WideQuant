class ArithAggregationNetwork(nn.Module):

   __init__(self, hidden_dim=768, n_heads=4, n_layers=2, n_types=3):
      arith_type_embedding: nn.Embedding(n_types, hidden_dim)
         Type indices: 0=TYPE_A, 1=TYPE_B, 2=TYPE_C
      transformer_encoder: nn.TransformerEncoder
         n_layers TransformerEncoderLayer each with:
         d_model=768, nhead=4, dim_feedforward=2048, dropout=0.1
      output_projection: nn.Linear(768, 768)

   forward(self, sub_quantity_embeddings: Tensor, arith_type: int) -> Tensor:
      sub_quantity_embeddings shape: (k, 768) where k = number of sub-quantities
      Step 1: type_emb = arith_type_embedding[arith_type].unsqueeze(0) → (1, 768)
      Step 2: x = cat([type_emb, sub_quantity_embeddings], dim=0) → (k+1, 768)
      Step 3: x = transformer_encoder(x.unsqueeze(1)).squeeze(1) → (k+1, 768)
      Step 4: pooled = x.mean(dim=0) → (768,)
      Step 5: return output_projection(pooled) → (768,)
      This output is y_resolved — drop-in replacement for y_b in DeepQuant scoring.

class AANTrainingObjective:

   compute_loss(self,
       aan_output: Tensor,          # (768,) AAN output for decomposed doc
       atomic_bert_output: Tensor   # (768,) BERT encoding of atomic doc quantity
   ) -> Tensor:
      L_AAN = MSELoss(aan_output, atomic_bert_output.detach())
      Use .detach() — treat atomic BERT encoding as fixed target.
      Return scalar loss.

   Else: print f"AAN QUALITY: FAIL — mean sim={mean:.3f}, need {threshold}"
